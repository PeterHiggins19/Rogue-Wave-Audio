    def isomap_tunnel(self, cluster: np.ndarray):
        n_cluster = len(cluster)
        if n_cluster < 10: return
        
        current_k = self.k[cluster]
        
        num_neighbors = max(5, min(20, int(5 + self.chi * 2)))
        
        k_norm = (current_k - current_k.min()) / (current_k.max() - current_k.min() + 1e-8)
        dist_local = np.abs(k_norm[:, np.newaxis] - k_norm[np.newaxis, :])
        
        adj = np.zeros((n_cluster, n_cluster))
        for i in range(n_cluster):
            neighbors = np.argsort(dist_local[i])[1:num_neighbors+1]
            adj[i, neighbors] = dist_local[i, neighbors]
            adj[neighbors, i] = dist_local[i, neighbors]
        
        geo_dist = shortest_path(adj, directed=False, method='FW')
        geo_dist[geo_dist == np.inf] = geo_dist.max() * 10
        
        H = np.eye(n_cluster) - np.ones((n_cluster, n_cluster)) / n_cluster
        G = -0.5 * H @ (geo_dist**2) @ H
        
        vals, vecs = np.linalg.eigh(G)
        vals = np.clip(vals, 0, None)
        idx = np.argsort(vals)[::-1]
        vals = vals[idx]
        vecs = vecs[:, idx]
        
        embed_dim = max(2, min(6, int(2 + self.chi * 0.6)))
        
        embedding = vecs[:, :embed_dim] @ np.diag(np.sqrt(vals[:embed_dim]))

    def tsne_tunnel(self, cluster: np.ndarray):
        n_cluster = len(cluster)
        if n_cluster < 10: return
        
        current_k = self.k[cluster]
        
        k_norm = (current_k - current_k.min()) / (current_k.max() - current_k.min() + 1e-8)
        
        perplexity = max(5.0, min(50.0, 5 + self.chi * 5))
        
        dist = squareform(pdist(k_norm.reshape(-1, 1)))
        
        P = np.zeros((n_cluster, n_cluster))
        for i in range(n_cluster):
            def perplexity_func(sigma):
                p = np.exp(-dist[i]**2 / (2 * sigma**2))
                p[i] = 0
                p /= p.sum() + 1e-8
                return 2**(-np.sum(p * np.log2(p + 1e-8)))
            
            sigma_min = 1e-8
            sigma_max = 1e3
            for _ in range(50):
                sigma = (sigma_min + sigma_max) / 2
                if perplexity_func(sigma) < perplexity:
                    sigma_min = sigma
                else:
                    sigma_max = sigma
            sigma = (sigma_min + sigma_max) / 2
            
            p = np.exp(-dist[i]**2 / (2 * sigma**2))
            p[i] = 0
            p /= p.sum() + 1e-8
            P[i] = p
        
        P = (P + P.T) / (2 * n_cluster)
        P = np.maximum(P, 1e-12)
        
        embed_dim = max(2, min(5, int(2 + self.chi * 0.4)))
        
        Y = np.random.normal(0, 1e-4, (n_cluster, embed_dim))
        
        momentum = 0.8
        eta = 500
        min_gain = 0.01
        gains = np.ones_like(Y)
        Y_prev = Y.copy()
        
        for iter in range(500):
            sum_Y = np.sum(Y**2, axis=1)
            num = 1 / (1 + sum_Y[:, np.newaxis] + sum_Y[np.newaxis, :] - 2 * Y @ Y.T)
            np.fill_diagonal(num, 0)
            Q = num / num.sum()
            Q = np.maximum(Q, 1e-12)
            
            PQ = P - Q
            dY = np.zeros_like(Y)
            for i in range(n_cluster):
                dY[i] = 4 * np.sum((PQ[:, i] - PQ[i, :])[:, np.newaxis] * (Y[i] - Y), axis=0)
            
            gains = (gains + 0.2) * ((dY * (Y - Y_prev) < 0)) + (gains * 0.8) * ((dY * (Y - Y_prev) >= 0))
            gains = np.clip(gains, min_gain, None)
            Y_prev = Y.copy()
            Y -= eta * gains * dY
            Y -= np.mean(Y, axis=0)
            
            if iter < 100:
                P_exag = P**4
                P_exag /= P_exag.sum()
        
        dist_embed = squareform(pdist(Y))
        close_threshold = np.percentile(dist_embed, 10)
        
        boost = np.zeros(n_cluster)
        for i in range(n_cluster):
            close = np.where(dist_embed[i] < close_threshold)[0]
            if len(close) > 1:
                tsne_boost = int(3 + self.chi * 0.6)
                boost[close] += tsne_boost
        
        self.k[cluster] += boost.astype(int)
        self.k[cluster] = np.clip(self.k[cluster], 10, 900)

    def umap_tunnel(self, cluster: np.ndarray):
        n_cluster = len(cluster)
        if n_cluster < 10: return
        
        current_k = self.k[cluster].copy()
        
        k_norm = (current_k - current_k.min()) / (current_k.max() - current_k.min() + 1e-8)
        
        n_neighbors = max(5, min(30, int(5 + self.chi * 3)))
        
        dist = squareform(pdist(k_norm.reshape(-1, 1)))
        
        knn_dists = np.partition(dist, n_neighbors, axis=1)[:, :n_neighbors]
        
        rho = np.min(knn_dists[:, 1:], axis=1) if n_neighbors > 1 else np.zeros(n_cluster)
        sigma = np.zeros(n_cluster)
        for i in range(n_cluster):
            d = knn_dists[i, 1:] - rho[i]
            d[d < 0] = 0
            if np.sum(d > 0) > 0:
                sigma[i] = np.exp(-d / d[d > 0].min())
        
        p_ij = np.exp(- (dist - rho[:, np.newaxis]) / sigma[:, np.newaxis])
        p_ij[dist > knn_dists[:, -1][:, np.newaxis]] = 0
        p_ij[np.arange(n_cluster), np.arange(n_cluster)] = 0
        p_ij = p_ij + p_ij.T - np.multiply(p_ij, p_ij.T)
        
        embed_dim = max(2, min(6, int(2 + self.chi * 0.5)))
        
        Y = np.random.normal(0, 1e-4, (n_cluster, embed_dim))
        
        a = 1.0
        b = 1.0
        epochs = 300 + int(self.chi * 50)
        
        for iter in range(epochs):
            attract = np.zeros_like(Y)
            for i in range(n_cluster):
                for j in range(n_cluster):
                    if p_ij[i,j] > 0:
                        d = Y[i] - Y[j]
                        dist_y = np.linalg.norm(d) + 1e-8
                        attract[i] += p_ij[i,j] * d / dist_y / (1 + a * dist_y**(2*b))
            
            repel = np.zeros_like(Y)
            q_ij = 1 / (1 + a * np.sum((Y[:, np.newaxis, :] - Y[np.newaxis, :, :])**2, axis=2)**(b))
            np.fill_diagonal(q_ij, 0)
            for i in range(n_cluster):
                for j in range(n_cluster):
                    if q_ij[i,j] > 0:
                        d = Y[i] - Y[j]
                        dist_y = np.linalg.norm(d) + 1e-8
                        repel[i] -= q_ij[i,j] * d / dist_y / (1 + a * dist_y**(2*b))
            
            Y += 0.1 * (attract + repel)
            Y -= np.mean(Y, axis=0)
        
        dist_embed = squareform(pdist(Y))
        close_threshold = np.percentile(dist_embed, 12)
        
        boost = np.zeros(n_cluster)
        for i in range(n_cluster):
            close = np.where(dist_embed[i] < close_threshold)[0]
            if len(close) > 1:
                umap_boost = int(3 + self.chi * 0.7)
                boost[close] += umap_boost
        
        self.k[cluster] += boost.astype(int)
        self.k[cluster] = np.clip(self.k[cluster], 10, 900)

        def diffusion_model_tunnel(self, cluster: np.ndarray):
            n_cluster = len(cluster)
            if n_cluster < 8: return
            
            current_k = self.k[cluster].copy().astype(np.float32)
            
            k_mean = current_k.mean()
            k_std = current_k.std() + 1e-8
            k_norm = (current_k - k_mean) / k_std
            
            num_steps = max(10, min(100, int(10 + self.chi * 10)))
            
            beta_start = 0.0001
            beta_end = 0.02
            betas = np.linspace(beta_start, beta_end, num_steps)
            alphas = 1.0 - betas
            alphas_cumprod = np.cumprod(alphas)
            
            t = num_steps - 1
            noise = np.random.normal(0, 1, n_cluster)
            k_noisy = np.sqrt(alphas_cumprod[t]) * k_norm + np.sqrt(1 - alphas_cumprod[t]) * noise
            
            for t_reverse in range(num_steps - 1, -1, -1):
                pred_noise = np.random.normal(0, np.sqrt(betas[t_reverse]), n_cluster)
                
                alpha_t = alphas[t_reverse]
                alpha_cum_t = alphas_cumprod[t_reverse]
                k_noisy = (k_noisy - np.sqrt(1 - alpha_cum_t) * pred_noise) / np.sqrt(alpha_t)
                
                if t_reverse > 0:
                    k_noisy += np.sqrt(betas[t_reverse]) * np.random.normal(0, 1, n_cluster)
            
            k_denoised = k_noisy * k_std + k_mean
            boost = (k_denoised - current_k) * (1 + self.chi / 2.0)
            
            self.k[cluster] += boost.astype(int)
            self.k[cluster] = np.clip(self.k[cluster], 10, 1000)
